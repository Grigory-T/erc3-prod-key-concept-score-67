def calculate_match_score(customer, search_terms):
    """Calculate a match score (0-1) for a customer against search terms"""
    if not customer:
        return 0
    
    score = 0
    max_possible = len(search_terms) * 2  # Each term can match in name and brief
    
    # Normalize strings for comparison
    name_lower = customer.name.lower() if customer.name else ""
    brief_lower = customer.brief.lower() if customer.brief else ""
    location_lower = customer.location.lower() if customer.location else ""
    
    for term in search_terms:
        term_lower = term.lower()
        
        # Check name

            score += 1
            # Bonus for exact match at word boundaries
            if term_lower == name_lower or f" {term_lower} " in f" {name_lower} ":

    # Normalize score to 0-1 range
    if max_possible > 0:
        normalized_score = min(score / max_possible, 1.0)
